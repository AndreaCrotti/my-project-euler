  (defun check_file (fname)
    (interactive)
    (if
        (file-exists-p fname)
        (concat "[[file:" fname "][X]]")
      ;; Otherwise create a link for a new file
      (concat "[[elisp:(find-file \"" fname "\")][_]]"))
    )
